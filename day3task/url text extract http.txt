  </head>
  <body>
    <script>
      //Create a new XMLHttpRequest object
      var xhr = new XMLHttpRequest();

      //Define HTTP method and URL
      var method = "GET";
      var url = "https://restcountries.com/v3.1/all";

      //Open the request
      xhr.open(method, url);

      xhr.onload = function () {
        if (xhr.status === 200) {
                    var details = JSON.parse(xhr.responseText);
                   for (var i = 0; i < details.length; i++) {
            console.log(details[i].name);console.log(details[i].population);console.log(details[i].region);console.log(details[i].subregion);
            
          }
        } else {
          //Request failed
          console.log("Request failed", xhr.status);
        }
      };

      //send the request
      xhr.send();
    </script>
  </body>
